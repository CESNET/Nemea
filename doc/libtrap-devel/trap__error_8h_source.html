57 /** Set error with default message.
58  *
59  * @param[in] ctx libtrap context
60  * @param[in] err_num Error number as defined in trap.h
61  * @return err_num
62  */
63 static inline int trap_error(trap_ctx_priv_t *ctx, int err_num)
64 {
65  if (ctx != NULL) {
66  ctx->trap_last_error = err_num;
67  if (err_num >= 0 &&
68  err_num < sizeof(default_err_msg)/sizeof(char*) &&
69  default_err_msg[err_num] != 0) {
70  ctx->trap_last_error_msg = default_err_msg[err_num];
71  }
72  else {
73  snprintf(ctx->error_msg_buffer, MAX_ERROR_MSG_BUFF_SIZE, "Unknown error (%i).", err_num);
74  ctx->trap_last_error_msg = ctx->error_msg_buffer;
75  }
76  }
77  return err_num;
78 }

81 /** Set error with custom message (printf-like formatting).
82  *
83  * @param[in] ctx libtrap context
84  * @param[in] err_num Error number as defined in trap.h
85  * @param[in] msg Human-readable string describing error, supports printf formatting.
86  * @param[in] ... Additional parameters for printf-like formatting of msg.
87  * @return err_num
88  */
89 static inline int trap_errorf(trap_ctx_priv_t *ctx, int err_num, const char *msg, ...)
90 {
91  if (pthread_rwlock_wrlock(&ctx->context_lock) != 0) {
92  VERBOSE(CL_ERROR, "%s: lock failed.", __func__);
93  return err_num;
94  }
95  ctx->trap_last_error = err_num;
96  va_list args;
97  va_start(args, msg);
98  vsnprintf(ctx->error_msg_buffer, MAX_ERROR_MSG_BUFF_SIZE, msg, args);
99  va_end(args);
100  ctx->trap_last_error_msg = ctx->error_msg_buffer;
101  pthread_rwlock_unlock(&ctx->context_lock);
102  return err_num;
103 }

105 /** Prepend given string before current ctx->trap_last_error_msg.
106  * This function is useful when a call of some function fails and you want
107  * to print a message about it but keep the original message about the error
108  * inside the function.
109  * Expected usage:
110  * return errorp("Call of myFunc failed: ");
111  *
112  * @param[in] ctx libtrap context
113  * @param[in] msg String to prepend current message
114  * @param[in] ... Additional parameters for printf-like formatting of msg.
115  * @return Current value of ctx->trap_last_error
116  */
117 static inline int trap_errorp(trap_ctx_priv_t *ctx, const char *msg, ...)
118 {
119  char tmp_str[MAX_ERROR_MSG_BUFF_SIZE];
120  strcpy(tmp_str, ctx->trap_last_error_msg);
121  va_list args;
122  va_start(args, msg);
123  int n = vsnprintf(ctx->error_msg_buffer, MAX_ERROR_MSG_BUFF_SIZE, msg, args);
124  va_end(args);
125  if (n >= 0 && n < MAX_ERROR_MSG_BUFF_SIZE) {
126  snprintf(ctx->error_msg_buffer + n, MAX_ERROR_MSG_BUFF_SIZE - n, tmp_str, args);
127  }
128  ctx->trap_last_error_msg = ctx->error_msg_buffer;
129  return ctx->trap_last_error;
130 }
