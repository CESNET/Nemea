172  * Disabling undesired versions of TLS/SSL and adding supported CAs to SSL_CTX.

174 static int tls_server_configure_ctx(const char *cert, SSL_CTX *ctx)
175 {
176  X509* certificate = X509_new();
177  BIO* bio_cert = BIO_new_file(cert, "r");
178 
179  PEM_read_bio_X509(bio_cert, &certificate, NULL, NULL);
180  if (certificate == NULL) {
181  VERBOSE(CL_ERROR, "Could not load certificate file.");
182  return EXIT_FAILURE;
183  }
184  if (SSL_CTX_add_client_CA(ctx, certificate) != 1) {
185  VERBOSE(CL_ERROR, "Could not add certificate to SSL_CTX.");
186  return EXIT_FAILURE;
187  }
188  /* disabling undesired versions of TLS */
189  SSL_CTX_set_options(ctx, SSL_OP_NO_SSLv2);
190  SSL_CTX_set_options(ctx, SSL_OP_NO_SSLv3);
191  SSL_CTX_set_options(ctx, SSL_OP_NO_TLSv1);
192  SSL_CTX_set_options(ctx, SSL_OP_NO_TLSv1_1);
193 
194  X509_free(certificate);
195  BIO_free_all(bio_cert);
196  return EXIT_SUCCESS;

281 /**
282  * Receive data
283  * \param[in] priv private IFC data
284  * \param[out] data received data
285  * \param[in,out] size expected size to wait for, it is used to return size that was not read
286  * \param[in] tm timeout
287  */
288 static int receive_part(void *priv, void **data, uint32_t *size, struct timeval *tm)
289 {
290  void *data_p = (*data);
291  tls_receiver_private_t *config = (tls_receiver_private_t *) priv;
292  ssize_t numbytes = *size;
293  int recvb, retval;
294  fd_set set;
295 
296  assert(data_p != NULL);
297 
298  while (config->is_terminated == 0) {
299  DEBUG_IFC(if (tm) {VERBOSE(CL_VERBOSE_LIBRARY, "Try to receive data in timeout %" PRIu64
300  "s%"PRIu64"us", tm->tv_sec, tm->tv_usec)});
301 
302  FD_ZERO(&set);
303  FD_SET(config->sd, &set);
304  /*
305  * Blocking or with timeout?
306  * With timeout 0,0 - non-blocking
307  */
308  retval = select(config->sd + 1, &set, NULL, NULL, tm);
309  if (retval > 0) {
310  if (FD_ISSET(config->sd, &set)) {
311  do {
312  recvb = SSL_read(config->ssl, data_p, numbytes);
313  if (recvb < 1) {
314  if (recvb == 0) {
315  errno = EPIPE;
316  }
317  switch (errno) {
318  case EINTR:
319  VERBOSE(CL_ERROR, "EINTR occured");
320  if (config->is_terminated == 1) {
321  client_socket_disconnect(priv);
322  return TRAP_E_TERMINATED;
323  }
324  break;
325  case EBADF:
326  case EPIPE:
327  client_socket_disconnect(priv);
328  return TRAP_E_IO_ERROR;
329  case EAGAIN:
330  (*size) = numbytes;
331  (*data) = data_p;
332  return TRAP_E_TIMEOUT;
333  }
334  }
335  numbytes -= recvb;
336  data_p += recvb;
337  DEBUG_IFC(VERBOSE(CL_VERBOSE_LIBRARY, "receive_part got %" PRId32 "B", recvb));
338  } while (numbytes > 0);
339  (*size) = numbytes;
340  (*data) = data_p;
341  return TRAP_E_OK;
342  } else {
343  continue;
344  }
345  } else if (retval == 0) {
346  VERBOSE(CL_VERBOSE_LIBRARY, "Timeout elapsed - non-blocking call used.");
347  (*size) = numbytes;
348  return TRAP_E_TIMEOUT;
349  } else if (retval < 0 && errno == EINTR) { /* signal received */
350  /** \todo continue with timeout minus time already waited */
351  VERBOSE(CL_VERBOSE_BASIC, "select interrupted");
352  continue;
353  } else { /* some error has occured */
354  VERBOSE(CL_VERBOSE_OFF, "select() returned %i (%s)", retval, strerror(errno));
355  client_socket_disconnect(priv);
356  return TRAP_E_IO_ERROR;
357  }
358  }
359  return TRAP_E_TERMINATED;
360 }

362 /**
363  * \brief Receive data from interface.
364  *
365  * It is expected that data is always the same pointer because it is buffer given by trap.c.
366  *
367  * This function contains finite state machine that controls receiving messages (header
368  * and payload), handles timeouts and sleep (to offload CPU during waiting for connection).
369  * The transition graph is:
370  * \dot
371  * digraph fsm { label="tls_receiver_recv()";labelloc=t;
372  * init -> conn_wait;
373  * init -> head_wait;
374  * init -> mess_wait;
375  * discard -> reset;
376  * reset -> init;
377  * reset -> init;
378  * reset -> reset;
379  * reset -> init;
380  * conn_wait -> reset;
381  * conn_wait -> head_wait;
382  * head_wait -> discard;
383  * head_wait -> reset;
384  * head_wait -> reset;
385  * head_wait -> mess_wait;
386  * mess_wait -> discard;
387  * mess_wait -> reset;
388  * }
389  * \enddot
390  *
391  * \param [in,out] priv private configuration structure
392  * \param [out] data where received data are stored
393  * \param [out] size size of received data
394  * \param [in] timeout timeout in usec, can be TRAP_WAIT, TRAP_HALFWAIT, or TRAP_NO_WAIT
395  * \return TRAP_E_OK (0) on success
396  */
397 int tls_receiver_recv(void *priv, void *data, uint32_t *size, int timeout)
398 {
399 #ifdef LIMITED_RECOVERY
400  uint32_t recovery = 0;
401 #endif
402  /** messageframe contains header that is read (even partially) in HEAD_WAIT */
403  trap_buffer_header_t messageframe;
404  tls_receiver_private_t *config = (tls_receiver_private_t *) priv;
405  void *p = &messageframe;
406  struct timeval tm, *temptm;
407  int retval;
408  /* first timestamp for global timeout in this function...
409  * in the RESET state, we should check the timeout given by caller
410  * with elapsed time from entry_time.
411  * Timeout is in microseconds... */
412  struct timespec spec_time;
413 
414  clock_gettime(CLOCK_MONOTONIC, &spec_time);
415  /* entry_time is in microseconds seconds -> secs * microsends + nanoseconds */
416  uint64_t entry_time = spec_time.tv_sec * 1000000 + (spec_time.tv_nsec / 1000);
417  uint64_t curr_time = 0;
418 
419  /* sleeptime (in usec) with sleeptimespec are used to wait
420  * for a while before next connecting when non-blocking. */
421  uint64_t sleeptime;
422  struct timespec sleeptimespec;
423 
424  /* correct module will pass only possitive timeout or TRAP_WAIT.
425  * TRAP_HALFWAIT is not valid value */
426  assert(timeout > TRAP_HALFWAIT);
427 
428  if ((config == NULL) || (data == NULL) || (size == NULL)) {
429  return TRAP_E_BAD_FPARAMS;
430  }
431  (*size) = 0;
432 
433  DEBUG_IFC(VERBOSE(CL_VERBOSE_LIBRARY, "recv trap_recv() was called"));
434 
435  /* convert libtrap timeout into timespec and timeval */
436  trap_set_timeouts(timeout, &tm, NULL);
437  temptm = (timeout==TRAP_WAIT?NULL:&tm);
438 
439  while (config->is_terminated == 0) {
440 init:
441  p = &messageframe;
442  DEBUG_IFC(VERBOSE(CL_VERBOSE_LIBRARY, "recv INIT"));
443  if (config->connected == 0) {
444  goto conn_wait;
445  } else {
446  if (config->data_pointer == NULL) {
447  goto head_wait;
448  } else {
449  /* continue where we timedout earlier */
450  p = config->data_pointer;
451  goto mess_wait;
452  }
453  }
454 discard:
455  DEBUG_IFC(VERBOSE(CL_VERBOSE_LIBRARY, "recv DISCARD"));
456  config->data_pointer = NULL;
457  goto reset;
458 reset:
459  if (config->is_terminated != 0) {
460  /* TRAP_E_TERMINATED is returned outside the loop */
461  break;
462  }
463  /* failure, next state is exit when we are non-blocking or INIT on blocking,
464  this state is a great place for handling timeouts. */
465  DEBUG_IFC(VERBOSE(CL_VERBOSE_LIBRARY, "recv RESET"));
466  if (timeout == TRAP_WAIT) {
467 #ifdef LIMITED_RECOVERY
468  if (++recovery > MAX_RECOVERY_TRY) {
469  goto init;
470  } else {
471  return TRAP_E_TIMEOUT;
472  }
473 #else
474  goto init;
475 #endif
476  } else {
477  /* non-blocking mode, let's check elapsed time */
478  clock_gettime(CLOCK_MONOTONIC, &spec_time);
479  curr_time = spec_time.tv_sec * 1000000 + (spec_time.tv_nsec / 1000);
480  if ((curr_time - entry_time) >= timeout) {
481  return TRAP_E_TIMEOUT;
482  } else {
483  if (config->connected == 0) {
484  /* wait at most 1 second before return to INIT */
485 
486  /* sleeptime is in usec */
487  sleeptime = timeout - (curr_time - entry_time);
488  /* if remaining sleeptime is higher than 1s, use 1s */
489  if (sleeptime < 1000000) {
490  sleeptimespec.tv_sec = sleeptime / 1000000;
491  sleeptimespec.tv_nsec = (sleeptime % 1000000) * 1000;
492  DEBUG_IFC(VERBOSE(CL_VERBOSE_LIBRARY, "sleep time set %" PRIu64
493  " us: %"PRIu64"s%"PRIu64"ns", sleeptime,
494  sleeptimespec.tv_sec, sleeptimespec.tv_nsec));
495  } else {
496  sleeptimespec.tv_sec = 1;
497  sleeptimespec.tv_nsec = 0;
498  }
499  /* We are not interested in reminder, because timeout will be
500  * checked again later. */
501  if (nanosleep(&sleeptimespec, NULL) == -1) {
502  if (errno == EINTR) {
503  goto reset;
504  } else {
505  VERBOSE(CL_ERROR, "recv nanosleep(): %s", strerror(errno));
506  }
507  }
508  DEBUG_IFC(VERBOSE(CL_VERBOSE_LIBRARY, "recv nanosleep finished"));
509  }
510 
511  /* update timeout that is used for recv after successful connection */
512  clock_gettime(CLOCK_MONOTONIC, &spec_time);
513  curr_time = spec_time.tv_sec * 1000000 + (spec_time.tv_nsec / 1000);
514  sleeptime = timeout - (int) (curr_time - entry_time);
515  if ((int) sleeptime > 0) {
516  trap_set_timeouts(sleeptime, &tm, NULL);
517  } else {
518  return TRAP_E_TIMEOUT;
519  }
520 
521  goto init;
522  }
523  }
524 conn_wait:
525  /* check if connected -> try to connect -> check if connected; next state is RESET or HEAD_WAIT */
526  /* expected next state is waiting for header */
527  DEBUG_IFC(VERBOSE(CL_VERBOSE_LIBRARY, "recv CONN_WAIT"));
528  if (config->connected == 0) {
529  /* we don't have connection, we must try to connect before accepting header */
530  retval = client_socket_connect(config, temptm);
531  if (retval == TRAP_E_FIELDS_MISMATCH) {
532  config->connected = 1;
533  return TRAP_E_FORMAT_MISMATCH;
534  } else if (retval == TRAP_E_OK) {
535  config->connected = 1;
536  /* ok, wait for header as we planned */
537  } else if (TRAP_E_BAD_CERT) {
538  /* TODO replace with TRAP_E_IO_ERROR and change it in test_echo_reply.c */
539  return TRAP_E_TERMINATED;
540  } else {
541  /* failed, reseting... */
542  if (timeout == TRAP_WAIT) {
543  /* Create a delay when blocking...
544  * This is specific situation, many attempts would be unpleasant */
545  sleep(1);
546  }
547  goto reset;
548  }
549  }
550  goto head_wait;
551 head_wait:
552  /* get and check header of message, next state can be MESS_WAIT or RESET */
553  DEBUG_IFC(VERBOSE(CL_VERBOSE_LIBRARY, "recv HEAD_WAIT (%p)", p));
554  config->data_wait_size = sizeof(trap_buffer_header_t);
555  retval = receive_part(config, &p, &config->data_wait_size, temptm);
556  if (retval != TRAP_E_OK) {
557  /* receiving failed */
558  DEBUG_IFC(VERBOSE(CL_VERBOSE_LIBRARY, "recv failed HEAD (%p) waiting %d B", p, config->data_wait_size));
559  if (retval == TRAP_E_IO_ERROR) {
560  /* disconnected -> drop data */
561  goto discard;
562  }
563  goto reset;
564  } else {
565  /* we expect to receive data */
566  messageframe.data_length = ntohl(messageframe.data_length);
567  config->data_wait_size = messageframe.data_length;
568  config->ext_buffer_size = messageframe.data_length;
569 #ifdef ENABLE_CHECK_HEADER
570  /* check if header is ok: */
571  if (tls_check_header(&messageframe) == 0) {
572  goto reset;
573  }
574 #endif
575  /* we got header, now we can start receiving payload */
576  p = data;
577  config->ext_buffer = data;
578  goto mess_wait;
579  }
580 mess_wait:
581  /* get and check payload of message, next state can be RESET or success exit */
582  /* receive payload */
583  DEBUG_IFC(VERBOSE(CL_VERBOSE_LIBRARY, "recv waiting MESS (%p) %d B", p, config->data_wait_size));
584  retval = receive_part(config, &p, &config->data_wait_size, temptm);
585  if (retval == TRAP_E_OK) {
586  /* Success! Data was already set by recv */
587  config->data_pointer = NULL;
588  (*size) = messageframe.data_length;
589  DEBUG_IFC(VERBOSE(CL_VERBOSE_LIBRARY, "recv get MESS (%p) remains: %d B", p, config->data_wait_size));
590  return TRAP_E_OK;
591  } else {
592  DEBUG_IFC(VERBOSE(CL_VERBOSE_LIBRARY, "recv get MESS (%p) still waiting for %d B", p, config->data_wait_size));
593  if (retval == TRAP_E_IO_ERROR) {
594  /* disconnected -> drop data */
595  goto discard;
596  }
597  config->data_pointer = p;
598  goto reset;
599  }
600  }
601  return TRAP_E_TERMINATED;
602 }

739  */
740 int create_tls_receiver_ifc(trap_ctx_priv_t *ctx, char *params, trap_input_ifc_t *ifc, uint32_t idx)
741 {

744  char *dest_addr = NULL;
745  char *dest_port = NULL;
746  char *keyfile = NULL;
747  char *certfile = NULL;
748  char *cafile = NULL;
749  tls_receiver_private_t *config = NULL;
750 
751  if (params == NULL) {
752  VERBOSE(CL_ERROR, "IFC requires at least three parameters (port:keyfile:certfile).");
753  return TRAP_E_BADPARAMS;
754  }
755 
756  config = (tls_receiver_private_t *) calloc(1, sizeof(tls_receiver_private_t));
757  if (config == NULL) {
758  VERBOSE(CL_ERROR, "Failed to allocate internal memory for input IFC.");
759  return TRAP_E_MEMORY;
760  }
761  config->ctx = ctx;
762  config->is_terminated = 0;
763  config->ifc_idx = idx;
764 
765  /* Parsing params */
766  param_iterator = trap_get_param_by_delimiter(params, &dest_addr, TRAP_IFC_PARAM_DELIMITER);
767  /* error! we expect 2 parameters */
768  if ((dest_addr == NULL) || (strlen(dest_addr) == 0)) {
769  VERBOSE(CL_ERROR, "Expected parameters: 'destination address:port:keyfile:certfile' are missing.");
770  result = TRAP_E_BADPARAMS;
771  goto failsafe_cleanup;
772  }
773  if (param_iterator != NULL) {
774  param_iterator = trap_get_param_by_delimiter(param_iterator, &dest_port, TRAP_IFC_PARAM_DELIMITER);
775  } else {
776  VERBOSE(CL_ERROR, "Missing 'dest_port', 'keyfile', 'certfile' and trusted 'CAfile' parameters.");
777  result = TRAP_E_BADPARAMS;
778  goto failsafe_cleanup;
779  }
780  if (param_iterator != NULL) {
781  param_iterator = trap_get_param_by_delimiter(param_iterator, &keyfile, TRAP_IFC_PARAM_DELIMITER);
782  } else {
783  VERBOSE(CL_ERROR, "Missing 'keyfile', 'certfile' and trusted 'CAfile' parameters.");
784  result = TRAP_E_BADPARAMS;
785  goto failsafe_cleanup;
786  }
787  if (param_iterator != NULL) {
788  param_iterator = trap_get_param_by_delimiter(param_iterator, &certfile, TRAP_IFC_PARAM_DELIMITER);
789  } else {
790  VERBOSE(CL_ERROR, "Missing 'certfile' and trusted 'CAfile' parameters.");
791  result = TRAP_E_BADPARAMS;
792  goto failsafe_cleanup;
793  }
794  if (param_iterator != NULL) {
795  param_iterator = trap_get_param_by_delimiter(param_iterator, &cafile, TRAP_IFC_PARAM_DELIMITER);
796  } else {
797  /* dest_addr skipped, move parameters */
798  cafile = certfile;
799  certfile = keyfile;
800  keyfile = dest_port;
801  dest_port = dest_addr;
802  dest_addr = strdup("localhost");
803  VERBOSE(CL_ERROR, "Only 3 parameters given, using 'localhost' as a destination address.");
804  }
805

807  config->int_mess_header.data_length = DEFAULT_MAX_DATA_LENGTH;
808  /* Parsing params ended */
809

811  config->dest_port = dest_port;

813  config->certfile = certfile;
814  config->cafile = cafile;

817  /* no delimiter found even if we expect two parameters */

820  goto failsafe_cleanup;
821  }
822

829  result = TRAP_E_MEMORY;
830  goto failsafe_cleanup;
831  }
832  if (tls_configure_ctx(config->sslctx, keyfile, certfile, cafile) == EXIT_FAILURE) {

834  goto failsafe_cleanup;

838  * In constructor, we do not know timeout yet.
839  * Use 5 seconds to wait for connection to output interface.
840  */
841 #ifndef ENABLE_NEGOTIATION
842  int retval = 0;
843  struct timeval tv = {5, 0};

848  VERBOSE(CL_VERBOSE_BASIC, "Could not connect to sender due to bad parameters.");
849  result = TRAP_E_BADPARAMS;

851  }
852  } else {
853  config->connected = 1;
854  }
855 #endif

858  ifc->recv = tls_receiver_recv;
859  ifc->destroy = tls_receiver_destroy;
860  ifc->terminate = tls_receiver_terminate;
861  ifc->create_dump = tls_receiver_create_dump;

863  ifc->get_id = tls_recv_ifc_get_id;
864  ifc->is_conn = tls_recv_ifc_is_conn;

866 #ifndef ENABLE_NEGOTIATION

868  VERBOSE(CL_VERBOSE_BASIC, "Could not connect to sender.");

870  result = retval;

872  }
873  }
874 #endif

876 failsafe_cleanup:

878  free(dest_port);
879  free(keyfile);

884  }
885  free(config);

911  * \param[in] tv timeout
912  * \return TRAP_E_OK on success, TRAP_E_TIMEOUT on error (can be caused by interrupt)
913  */
914 static int wait_for_connection(int sock, struct timeval *tv)
915 {

917  fd_set fdset;
918  FD_ZERO(&fdset);
919  FD_SET(sock, &fdset);
920  VERBOSE(CL_VERBOSE_LIBRARY, "wait for connection");
921 
922  rv = select(sock + 1, NULL, &fdset, NULL, tv);
923  if (rv == 1) {
924  int so_error;
925  socklen_t len = sizeof so_error;
926 
927  getsockopt(sock, SOL_SOCKET, SO_ERROR, &so_error, &len);
928 
929  if (so_error == 0) {
930  return TRAP_E_OK;
931  }
932  }
933  return TRAP_E_TIMEOUT;

937 /**
938  * \brief client_socket is used as a receiver
939  * \param[in] c pointer to module private data
940  * \param[in] tv timeout
941  * \return TRAP_E_OK on success
942  */
943 static int client_socket_connect(tls_receiver_private_t *c, struct timeval *tv)
944 {
945  int sockfd = -1, options;
946  union tls_socket_addr addr;
947  struct addrinfo *servinfo, *p = NULL;
948  int rv, addr_count = 0;
949  char s[INET6_ADDRSTRLEN];
950 
951  if ((c == NULL) || (c->dest_addr == NULL) || (c->dest_port == NULL)) {
952  return TRAP_E_BAD_FPARAMS;
953  }
954 
955  memset(&addr, 0, sizeof(addr));
956 
957  addr.tls_addr.ai_family = AF_UNSPEC;
958  addr.tls_addr.ai_socktype = SOCK_STREAM;
959 
960  if ((rv = getaddrinfo(c->dest_addr, c->dest_port, &addr.tls_addr, &servinfo)) != 0) {
961  VERBOSE(CL_ERROR, "getaddrinfo: %s", gai_strerror(rv));
962  return TRAP_E_IO_ERROR;
963  }
964 
965  DEBUG_IFC(VERBOSE(CL_VERBOSE_LIBRARY, "recv Try to connect"));
966 
967  if (tv != NULL) {
968  /* compute uniform intervals for all possible address */
969  for (p = servinfo; p != NULL; p = p->ai_next) {
970  addr_count++;
971  }
972  tv->tv_sec = (tv->tv_sec * 1000000 + tv->tv_usec) / addr_count;
973  tv->tv_usec = tv->tv_sec % 1000000;
974  tv->tv_sec /= 1000000;
975  VERBOSE(CL_VERBOSE_LIBRARY, "Every address will be tried for timeout: %"PRId64"s%"PRId64"us",
976  tv->tv_sec, tv->tv_usec);
977  }
978 
979  /* loop through all the results and connect to the first we can */
980  for (p = servinfo; p != NULL; p = p->ai_next) {
981  if ((sockfd = socket(p->ai_family, p->ai_socktype, p->ai_protocol)) == -1) {
982  continue;
983  }
984  if ((options = fcntl(sockfd, F_GETFL)) != -1) {
985  if (fcntl(sockfd, F_SETFL, O_NONBLOCK | options) == -1) {
986  VERBOSE(CL_ERROR, "Could not set socket to non-blocking.");
987  }
988  }
989  if (connect(sockfd, p->ai_addr, p->ai_addrlen) == -1) {
990  if (errno != EINPROGRESS && errno != EAGAIN) {
991  DEBUG_IFC(VERBOSE(CL_VERBOSE_LIBRARY, "recv TCPIP ifc connect error %d (%s)", errno,
992  strerror(errno)));
993  close(sockfd);
994  sockfd = -1;
995  continue;
996  } else {
997  rv = wait_for_connection(sockfd, tv);
998  if (rv == TRAP_E_TIMEOUT) {
999  if (c->is_terminated) {
1000  rv = TRAP_E_TERMINATED;
1001  break;
1002  }
1003  /* try another address */
1004  close(sockfd);
1005  sockfd = -1;
1006  continue;
1007  } else {
1008  /* success */
1009  rv = TRAP_E_OK;
1010  break;
1011  }
1012  }
1013  }
1014  break;
1015  }
1016  /* there was no successfull connection for whole servinfo struct */
1017  if (p == NULL) {
1018  VERBOSE(CL_VERBOSE_LIBRARY, "recv client: Connection failed.");
1019  rv = TRAP_E_TIMEOUT;
1020  }
1021 
1022  /* catching all possible errors from setting up socket before atempting tls handshake */
1023  if (rv != TRAP_E_OK) {
1024  freeaddrinfo(servinfo);
1025  close(sockfd);
1026  return rv;
1027  }
1028 
1029  if (p != NULL) {
1030  if (inet_ntop(p->ai_family, get_in_addr((struct sockaddr *)p->ai_addr), s, sizeof s) != NULL) {
1031  VERBOSE(CL_VERBOSE_LIBRARY, "recv client: connected to %s", s);
1032  }
1033  }
1034  freeaddrinfo(servinfo);
1035  servinfo = NULL;
1036 
1037  c->sd = sockfd;
1038  c->ssl = SSL_new(c->sslctx);
1039  if (c->ssl == NULL) {
1040  VERBOSE(CL_ERROR, "Creating SSL structure failed: %s", ERR_reason_error_string(ERR_get_error()));
1041  return TRAP_E_MEMORY;
1042  }
1043 
1044  /* setting tcp socket to be used for ssl connection */
1045  if (SSL_set_fd(c->ssl, c->sd) != 1) {
1046  VERBOSE(CL_ERROR, "Setting SSL file descriptor to tcp socket failed: %s",
1047  ERR_reason_error_string(ERR_get_error()));
1048  return TRAP_E_IO_ERROR;
1049  }
1050  SSL_set_connect_state(c->ssl);
1051 
1052  do {
1053  rv = SSL_connect(c->ssl);
1054  if (rv < 1) {
1055  rv = ERR_get_error();
1056  switch (rv) {
1057  case SSL_ERROR_NONE:
1058  case SSL_ERROR_WANT_CONNECT:
1059  case SSL_ERROR_WANT_X509_LOOKUP:
1060  case SSL_ERROR_WANT_READ:
1061  case SSL_ERROR_WANT_WRITE:
1062  break;
1063  default:
1064  VERBOSE(CL_ERROR, "SSL connection failed, could be wrong certificate. %s",
1065  ERR_reason_error_string(ERR_get_error()));
1066  SSL_free(c->ssl);
1067  c->ssl = NULL;
1068  close(c->sd);
1069  return TRAP_E_IO_ERROR;
1070  }
1071  }
1072  } while (rv < 1);
1073  VERBOSE(CL_VERBOSE_BASIC, "SSL successfully connected")
1074 
1075  int ret_ver = verify_certificate(c->ssl); /* server certificate verification */
1076  if (ret_ver != 0){
1077  VERBOSE(CL_VERBOSE_LIBRARY, "verify_certificate: failed to verify server's certificate");
1078  SSL_free(c->ssl);
1079  c->ssl = NULL;
1080  return TRAP_E_BAD_CERT;
1081  }
1082 
1083  /** Input interface negotiation */
1084 #ifdef ENABLE_NEGOTIATION
1085  switch(input_ifc_negotiation(c, TRAP_IFC_TYPE_TLS)) {
1086  case NEG_RES_FMT_UNKNOWN:
1087  VERBOSE(CL_VERBOSE_LIBRARY, "Input_ifc_negotiation result: failed (unknown data format of the output interface).");
1088  close(sockfd);
1089  return TRAP_E_TIMEOUT;
1090 
1091  case NEG_RES_CONT:
1092  VERBOSE(CL_VERBOSE_LIBRARY, "Input_ifc_negotiation result: success.");
1093  return TRAP_E_OK;
1094 
1095  case NEG_RES_FMT_CHANGED: /* used on format change with JSON */
1096  VERBOSE(CL_VERBOSE_LIBRARY, "Input_ifc_negotiation result: success (format has changed; it was not first negotiation).");
1097  return TRAP_E_OK;
1098 
1099  case NEG_RES_RECEIVER_FMT_SUBSET: /* used on format change with UniRec */
1100  VERBOSE(CL_VERBOSE_LIBRARY, "Input_ifc_negotiation result: success (required set of fields of the input interface is subset of the recevied format).");
1101  return TRAP_E_OK;
1102 
1103  case NEG_RES_SENDER_FMT_SUBSET: /* used on format change with UniRec */
1104  VERBOSE(CL_VERBOSE_LIBRARY, "Input_ifc_negotiation result: success (new recevied format specifier is subset of the old one; it was not first negotiation).");
1105  return TRAP_E_OK;
1106 
1107  case NEG_RES_FAILED:
1108  VERBOSE(CL_VERBOSE_LIBRARY, "Input_ifc_negotiation result: failed (error while receiving hello message from output interface).");
1109  return TRAP_E_FIELDS_MISMATCH;
1110 
1111  case NEG_RES_FMT_MISMATCH:
1112  VERBOSE(CL_VERBOSE_LIBRARY, "Input_ifc_negotiation result: failed (data type or data format specifier mismatch).");
1113  return TRAP_E_FIELDS_MISMATCH;
1114 
1115  default:
1116  VERBOSE(CL_VERBOSE_LIBRARY, "Input_ifc_negotiation result: default case");
1117  break;
1118  }
1119 #endif
1120 
1121 
1122  return TRAP_E_OK;
1123 }

1147 /**
1148  * \brief Try to send data block at once
1149  *
1150  * \param [in] c private data
1151  * \param [in] cl pointer to client structure
1152  * \param [in,out] data pointer to beginning of data
1153  * \param [in,out] size size of data to send and the rest unsent size of data
1154  * \param [in] block 1 if blocking, 0 if non-blocking
1155  * \return TRAP_E_OK, TRAP_E_TIMEOUT, TRAP_E_TERMINATED, TRAP_E_IO_ERROR
1156  */
1157 static int send_all_data(tls_sender_private_t *c, struct tlsclient_s *cl, void **data, uint32_t *size, char block)
1158 {
1159  void *p = (*data);
1160  ssize_t numbytes = (*size), sent_b;
1161  int res = TRAP_E_TERMINATED;
1162 
1163 again:
1164  sent_b = SSL_write(cl->ssl, p, numbytes);
1165  if (sent_b <= 0) {
1166  switch (SSL_get_error(cl->ssl, sent_b)) {
1167  case SSL_ERROR_ZERO_RETURN:
1168  case SSL_ERROR_SYSCALL:
1169  VERBOSE(CL_VERBOSE_OFF, "Disconnected client (%i)", errno);
1170  res = TRAP_E_IO_ERROR;
1171  goto failure;
1172 
1173  case SSL_ERROR_WANT_READ:
1174  case SSL_ERROR_WANT_WRITE:
1175  if (block == 1) {
1176  usleep(NONBLOCKING_MINWAIT);
1177  goto again;
1178  }
1179  break;
1180  }
1181  if (c->is_terminated == 1) {
1182  res = TRAP_E_TERMINATED;
1183  goto failure;
1184  }
1185  } else if (sent_b > 0) {
1186  numbytes -= sent_b;
1187  p += sent_b;
1188  DEBUG_IFC(VERBOSE(CL_VERBOSE_LIBRARY, "send sent: %"PRId64" B remaining: "
1189  "%"PRIu64" B from %p (errno %"PRId32")",
1190  sent_b, numbytes, p, errno));
1191  }
1192  assert(numbytes>=0);
1193  (*size) = numbytes;
1194  if (numbytes > 0) {
1195  if (block == 1) {
1196  goto again;
1197  }
1198  (*data) = p;
1199  return TRAP_E_TIMEOUT;
1200  } else if (numbytes == 0) {
1201  (*data) = NULL;
1202  return TRAP_E_OK;
1203  }
1204 failure:
1205  (*data) = NULL;
1206  (*size) = 0;
1207  return res;
1208 }

1231 /**
1232  * \brief Check if we have connected clients and wait for them.
1233  *
1234  * The function is blocking or non-blocking.
1235  * \param [in] config private data
1236  * \param [in] t pointer to timeout for sem_timedwait(), NULL when blocking
1237  * \return TRAP_E_OK - we have clients, TRAP_E_TIMEOUT - we don't have clients, TRAP_E_TERMINATED - terminating
1238  */

1240 {

1242  assert(t != NULL);
1243

1245  /* there is no connected client */

1247  if (res == -1) {

1249  DEBUG_IFC(VERBOSE(CL_ERROR, "interrupt."));
1250  if (config->is_terminated != 0) {
1251  return TRAP_E_TERMINATED;
1252  }
1253  } else if (errno == ETIMEDOUT) {
1254  return TRAP_E_TIMEOUT; /* no client after timeout */
1255  } else {
1256  VERBOSE(CL_ERROR, "sem_timedwait failed (%d): %s", errno, strerror(errno));
1257  }
1258  }
1259  /* timeout or connected client */
1260  if (check_connected_clients(config) == 0) {
1261  return TRAP_E_TIMEOUT; /* no client after timeout */
1262  }
1263  }
1264  /* connected client */
1265  return TRAP_E_OK;
1266 }

1268 /**
1269  * \brief Send data to all connected clients.
1270  *
1271  * All clients are in 'current' buffer or in 'backup' buffer.
1272  * All clients get messages from current buffer at first.
1273  * * When all buffer is successfully sent to all clients
1274  * in 'current' buffer, everything is alright.
1275  *
1276  * * When everybody fails due to timeout, everything
1277  * is alright, everybody stays in 'current' buffer.
1278  *
1279  * * When somebody fails and somebody is successful, failing clients are moved into 'backup' buffer and they have chance to receive the rest of message later. We disconnect all clients that were in 'backup' buffer before moving new ones from 'current' buffer.
1280  *
1281  * * All successful clients from 'backup' buffer are moved back into 'current' buffer.
1282  *
1283  * \param[in] priv pointer to module private data
1284  * \param[in] data pointer to data to send
1285  * \param[in] size size of data to send
1286  * \param[in] timeout timeout in microseconds
1287  * \return 0 on success (TRAP_E_OK), TRAP_E_BAD_FPARAMS if sender was not properly initialized, TRAP_E_TERMINATED if interface was terminated.
1288  */
1289 int tls_sender_send(void *priv, const void *data, uint32_t size, int timeout)
1290 {
1291  uint8_t buffer[DEFAULT_MAX_DATA_LENGTH];
1292  int result = TRAP_E_TIMEOUT;
1293  tls_sender_private_t *c = (tls_sender_private_t *) priv;
1294  /* timeout for select */
1295  struct timeval tv;
1296  /* timeout for sem_timedwait */
1297  struct timespec ts = { .tv_sec = 0, .tv_nsec = 0 };
1298  fd_set set, disset;
1299  int maxsd = -1;
1300  struct tlsclient_s *cl;
1301  uint32_t i, j, failed, passed;
1302  int retval;
1303  ssize_t readbytes;
1304 
1305  char block = ((timeout == TRAP_WAIT || timeout == TRAP_HALFWAIT) ? 1 : 0);
1306 
1307  /* pointer to timeout for select() */
1308  struct timeval *tv_p = ((block != 0) ? NULL : &tv);
1309  /* pointer to timeout for sem_timedwait() */
1310  struct timespec *ts_p = &ts;
1311 
1312  /* correct module will pass only possitive timeout or TRAP_WAIT, TRAP_HALFWAIT */
1313  assert(timeout >= TRAP_HALFWAIT);
1314 
1315  /* I. Init phase: set timeout and double-send switch */
1316 
1317 blocking_repeat:
1318 
1319  if (c->is_terminated) {
1320  return TRAP_E_TERMINATED;
1321  }
1322 
1323  switch (timeout) {
1324  case TRAP_WAIT:
1325  trap_set_timeouts(1000000, &tv, &ts);
1326  break;
1327  case TRAP_HALFWAIT:
1328  /*
1329  * wait 1s in a loop for select(),
1330  * do not change timeout for sem_timedwait in connphase
1331  */
1332  trap_set_timeouts(1000000, &tv, NULL);
1333  break;
1334  default:
1335  /*
1336  * set timeout (can be 0 - nowait or any positive number) for select(),
1337  * do not change timeout for sem_timedwait in connphase
1338  */
1339  trap_set_timeouts(timeout, &tv, NULL);
1340  break;
1341  }
1342 
1343  /* Check connected clients and wait for them when blocking */
1344  result = tls_sender_conn_phase(c, ts_p);
1345  if (result != TRAP_E_OK) {
1346  /* it is useless to send when nobody receives */
1347  goto exit;
1348  }
1349 
1350  FD_ZERO(&disset);
1351  FD_ZERO(&set);
1352 
1353  /*
1354  * add term_pipe for reading into the disconnect client set
1355  */
1356  FD_SET(c->term_pipe[0], &disset);
1357  if (maxsd < c->term_pipe[0]) {
1358  maxsd = c->term_pipe[0];
1359  }
1360 
1361  for (i = 0, j = 0; i < c->clients_arr_size; ++i) {
1362  cl = &c->clients[i];
1363  if (j == c->connected_clients) {
1364  break;
1365  }
1366  if (cl->sd <= 0) {
1367  /* not connected client */
1368  continue;
1369  }
1370  j++;
1371  if (maxsd < cl->sd) {
1372  maxsd = cl->sd;
1373  }
1374  /* check if client is still connected */
1375  FD_SET(cl->sd, &disset);
1376  if (cl->client_state != TLSCURRENT_COMPLETE) {
1377  /* check if client is ready for message */
1378  FD_SET(cl->sd, &set);
1379  }
1380  }
1381 
1382  retval = select(maxsd + 1, &disset, &set, NULL, tv_p);
1383  if (retval == 0) {
1384  if (block == 0) {
1385  /* non-blocking mode */
1386  result = TRAP_E_TIMEOUT;
1387  goto exit;
1388  } else {
1389  /* blocking mode */
1390  goto blocking_repeat;
1391  }
1392  } else if (retval < 0) {
1393  if (c->is_terminated != 0) {
1394  goto exit;
1395  } else if (errno == EBADF) {
1396  assert(0);
1397  result = TRAP_E_IO_ERROR;
1398  goto exit;
1399  }
1400  }
1401 
1402  passed = failed = 0;
1403 
1404  pthread_mutex_lock(&c->sending_lock);
1405  for (i = 0, j = 0; i < c->clients_arr_size; ++i) {
1406  if (j == c->connected_clients) {
1407  break;
1408  }
1409  cl = &c->clients[i];
1410  if(cl->sd == -1) {
1411  continue;
1412  }
1413  if (FD_ISSET(cl->sd, &disset)) {
1414  /* client disconnects */
1415  readbytes = recv(cl->sd, buffer, DEFAULT_MAX_DATA_LENGTH, MSG_NOSIGNAL | MSG_DONTWAIT);
1416  if (readbytes < 1) {
1417  VERBOSE(CL_VERBOSE_LIBRARY, "Disconnected client.");
1418  result = TRAP_E_IO_ERROR;
1419  server_disconnected_client(c, i);
1420  continue;
1421  }
1422  }
1423 
1424  if (FD_ISSET(cl->sd, &set)) {
1425  if (j == c->connected_clients) {
1426  break;
1427  }
1428  /* we added only clients whose sending is not TLSCURRENT_COMPLETE */
1429  if ((cl->sending_pointer == NULL) || (cl->pending_bytes == 0)) {
1430  cl->sending_pointer = (void *) data;
1431  cl->pending_bytes = size;
1432  }
1433  result = send_all_data(c, cl, &cl->sending_pointer, &cl->pending_bytes, block);
1434  switch (result) {
1435  case TRAP_E_IO_ERROR:
1436  server_disconnected_client(c, i);
1437  failed++;
1438  break;
1439  case TRAP_E_OK:
1440  passed++;
1441  cl->client_state = TLSCURRENT_COMPLETE;
1442  break;
1443  case TRAP_E_TIMEOUT:
1444  failed++;
1445  break;
1446  }
1447  j++;
1448  }
1449  if (c->connected_clients == 0) {
1450  /* there is no client to send to */
1451  result = TRAP_E_IO_ERROR;
1452  break;
1453  }
1454  }
1455  pthread_mutex_unlock(&c->sending_lock);
1456 
1457  if (FD_ISSET(c->term_pipe[0], &disset)) {
1458  /* Sending was interrupted by terminate(), exit even from TRAP_WAIT function call. */
1459  return TRAP_E_TERMINATED;
1460  }
1461 
1462  if (failed != 0) {
1463  result = TRAP_E_TIMEOUT;
1464  } else {
1465  for (i = 0, passed = 0; i < c->clients_arr_size; ++i) {
1466  cl = &c->clients[i];
1467  if ((cl->sd > 0) && (cl->client_state == TLSCURRENT_COMPLETE) && (cl->sending_pointer == NULL)) {
1468  passed++;
1469  if (passed == c->connected_clients) {
1470  break;
1471  }
1472  }
1473  }
1474  if (passed == c->connected_clients) {
1475  /* there is no client that failed */
1476  for (i = 0, j = 0; i < c->clients_arr_size; ++i) {
1477  cl = &c->clients[i];
1478  if ((cl->sd > 0) && (cl->client_state == TLSCURRENT_COMPLETE)) {
1479  cl->client_state = TLSCURRENT_IDLE;
1480  j++;
1481  if (j == passed) {
1482  break;
1483  }
1484  }
1485  }
1486  result = TRAP_E_OK;
1487  } else {
1488  if (block != 0) {
1489  goto blocking_repeat;
1490  }
1491  }
1492  }
1493 
1494 exit:
1495  /*
1496  * Return to blocking_repeat ONLY when the timeout is TRAP_WAIT.
1497  * TRAP_HALFWAIT is handled before.
1498  */
1499  if ((timeout == TRAP_WAIT) && ((result != TRAP_E_OK) || (c->connected_clients == 0))) {
1500  goto blocking_repeat;
1501  }
1502  return result;
1503 }
