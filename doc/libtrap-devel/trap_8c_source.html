224 /**
225  * \brief Check content of buffer, iterate over message headers
226  * \param [in] buffer start of buffer
227  * \param [in] buffer_size size of buffer
228  * \return 0 on success, number of errors otherwise
229  */
230 int trap_check_buffer_content(void *buffer, uint32_t buffer_size)
231 {
232  uint32_t offset, check_mess_counter = 0;
233  uint16_t *check_mess_header;
234  int errors = 0;
235  void *check_mess_pointer;
236  for (offset = 0, check_mess_header = check_mess_pointer = buffer;
237  ((offset < buffer_size) && (offset < TRAP_IFC_MESSAGEQ_SIZE));) {
238  check_mess_counter++;
239  /* go to next size, skip header + payload */
240  offset += sizeof(*check_mess_header) + (*check_mess_header);
241  check_mess_pointer += sizeof(*check_mess_header) + (*check_mess_header);
242  check_mess_header = (uint16_t *) check_mess_pointer;
243  }
244  if (offset != buffer_size) {
245  VERBOSE(CL_ERROR, "Not enough data or some headers are malformed.");
246  errors++;
247  return errors;
248  }
249  return errors;
250 }

261 static inline int trap_read_from_buffer(trap_ctx_priv_t *ctx, uint32_t ifc_idx, const void **data, uint16_t *size, int timeout)
262 {
263  int result = TRAP_E_TIMEOUT;
264  /* pointer to current message header */
265  uint32_t tempbufheader = 0;
266 
267  /* pointer to current message payload */
268  void *bp = ctx->in_ifc_list[ifc_idx].buffer;
269  pthread_mutex_lock(&ctx->in_ifc_list[ifc_idx].ifc_mtx);
270  if ((ctx->in_ifc_list[ifc_idx].buffer_full == 0) || (ctx->in_ifc_list[ifc_idx].buffer_full > TRAP_IFC_MESSAGEQ_SIZE)) {
271  /* get new data and store into buffer, set buffer_full size */
272  ctx->in_ifc_list[ifc_idx].buffer_pointer = ctx->in_ifc_list[ifc_idx].buffer;
273  result = ctx->in_ifc_list[ifc_idx].recv(ctx->in_ifc_list[ifc_idx].priv, bp, &tempbufheader, timeout);
274  if (result == TRAP_E_FORMAT_MISMATCH) {
275  goto exit;
276  }
277 #ifdef BUFFERING_CHECK_HEADERS
278  if (trap_check_buffer_content(bp, tempbufheader) != 0) {
279  VERBOSE(CL_ERROR, "Buffer is not valid.");
280  }
281 #endif
282  if (result == TRAP_E_OK) {
283  ctx->counter_recv_buffer[ifc_idx]++;
284 
285  ctx->in_ifc_list[ifc_idx].buffer_full = tempbufheader;
286  ctx->in_ifc_list[ifc_idx].buffer_pointer = ctx->in_ifc_list[ifc_idx].buffer;
287  DEBUG_BUF(VERBOSE(CL_VERBOSE_LIBRARY, "read received new buffer new bf %"PRIu32" %p",
288  ctx->in_ifc_list[ifc_idx].buffer_full,
289  ctx->in_ifc_list[ifc_idx].buffer_pointer));
290  #ifdef TESTBUFFERING
291  VERBOSE(CL_VERBOSE_OFF, "Received buffer of size %u.", ctx->in_ifc_list[ifc_idx].buffer_full);
292  #endif
293  } else {
294  goto exit;
295  }
296  }
297 
298  if (ctx->in_ifc_list[ifc_idx].buffer_full > 0) {
299  /* get message from buffer */
300  (*size) = ntohs(*((uint16_t *) ctx->in_ifc_list[ifc_idx].buffer_pointer));
301  (*data) = (ctx->in_ifc_list[ifc_idx].buffer_pointer + sizeof(*size));
302  /* decrease buffer_full size by returned payload and its header */
303  ctx->in_ifc_list[ifc_idx].buffer_full -= (*size + sizeof(*size));
304  ctx->in_ifc_list[ifc_idx].buffer_pointer += (*size) + sizeof(*size);
305  DEBUG_BUF(VERBOSE(CL_VERBOSE_LIBRARY, "read from buffer %"PRIu16" B skip %"PRIu64" B, new bf %"PRIu32" %p",
306  (*size),
307  (*size + sizeof(*size)),
308  ctx->in_ifc_list[ifc_idx].buffer_full,
309  ctx->in_ifc_list[ifc_idx].buffer_pointer));
310  result = TRAP_E_OK;
311  } else {
312  (*size) = 0;
313  }
314 exit:
315  pthread_mutex_unlock(&ctx->in_ifc_list[ifc_idx].ifc_mtx);
316  if (result == TRAP_E_OK) {
317  ctx->counter_recv_message[ifc_idx]++;
318  if (ctx->in_ifc_list[ifc_idx].client_state == FMT_CHANGED) {
319  ctx->in_ifc_list[ifc_idx].client_state = FMT_OK;
320  return TRAP_E_FORMAT_CHANGED;
321  }
322  }
323  return result;
324 }

336 static inline int trap_store_into_buffer(trap_ctx_priv_t *ctx, unsigned int ifc, const void *data, uint16_t size, int timeout, char flush)
337 {
338  /* Declaration of variables, we can have small buffer, initialization after checking the condition. */
339  uint32_t freespace, needed_size = size + sizeof(size);
340  int result;
341 
342  if (ctx->out_ifc_list[ifc].ifc_type == TRAP_IFC_TYPE_BLACKHOLE) {
343  return TRAP_E_OK;
344  }
345 
346  /* Can we put message at least into empty buffer? In the worst case, we could end up with SEGFAULT -> rather skip with error */
347  if (needed_size > TRAP_IFC_MESSAGEQ_SIZE) {
348  return trap_errorf(ctx, TRAP_E_MEMORY, "Buffer is too small for this message. Skipping...");
349  }
350 
351  if (flush != 0) {
352  /* Autoflush call, trying to lock section, maybe interface is waiting for clients -> rather skip than block the whole thread. */
353  if (pthread_mutex_trylock(&ctx->out_ifc_list[ifc].ifc_mtx) != 0) {
354  return TRAP_E_OK;
355  }
356  } else {
357  /* Lock this section at first before sending whole buffer. */
358  pthread_mutex_lock(&ctx->out_ifc_list[ifc].ifc_mtx);
359  }
360  /* initialization in locked section, otherwise autoflush can send buffer which has been already sent */
361  if (ctx->out_ifc_list[ifc].buffer_index <= (TRAP_IFC_MESSAGEQ_SIZE - sizeof(trap_buffer_header_t))) {
362  freespace = TRAP_IFC_MESSAGEQ_SIZE - ctx->out_ifc_list[ifc].buffer_index - sizeof(trap_buffer_header_t);
363  } else {
364  freespace = 0;
365  }
366  result = TRAP_E_TIMEOUT;
367 
368  /* Is this a autoflush call? If we have empty buffer, we do not send anything. */
369  if (flush != 0) {
370  if (ctx->out_ifc_list[ifc].buffer_index != 0) {
371 #ifdef BUFFERING_CHECK_HEADERS
372  if (trap_check_buffer_content(ctx->out_ifc_list[ifc].buffer, ctx->out_ifc_list[ifc].buffer_index) != 0) {
373  VERBOSE(CL_ERROR, "Buffer is not valid.");
374  }
375 #endif
376  DEBUG_BUF(VERBOSE(CL_VERBOSE_LIBRARY, "sending by autoflush %"PRIu32" B from %p", ctx->out_ifc_list[ifc].buffer_index, ctx->out_ifc_list[ifc].buffer));
377 
378  ctx->out_ifc_list[ifc].buffer_occupied = 1;
379  trap_buffer_header_t *h = (trap_buffer_header_t *) ctx->out_ifc_list[ifc].buffer_header;
380  h->data_length = htonl(ctx->out_ifc_list[ifc].buffer_index);
381  result = ctx->out_ifc_list[ifc].send(ctx->out_ifc_list[ifc].priv, ctx->out_ifc_list[ifc].buffer_header,
382  ctx->out_ifc_list[ifc].buffer_index + sizeof(trap_buffer_header_t), timeout);
383 
384  if (result == TRAP_E_OK) {
385  ctx->counter_send_buffer[ifc]++;
386  ctx->out_ifc_list[ifc].buffer_index = 0;
387  ctx->out_ifc_list[ifc].buffer_occupied = 0;
388  DEBUG_BUF(VERBOSE(CL_VERBOSE_LIBRARY, "Sending partial buffer invoked by autoflush timeout on interface %d", ifc));
389  } else {
390  VERBOSE(CL_VERBOSE_LIBRARY, "Autoflush was not successful.");
391  if (trap_ctx_get_client_count(ctx, ifc) == 0) {
392  ctx->out_ifc_list[ifc].buffer_occupied = 0;
393  }
394  }
395  }
396  goto fn_exit;
397  }
398  /* we send buffer before timeout, no need to flush it */
399  ctx->out_ifc_list[ifc].bufferflush = 1;
400 
401  if ((freespace >= needed_size) && (ctx->out_ifc_list[ifc].bufferswitch == 1)) {
402  /* we have enough space, buffering is enabled and size is not "flush" */
403 
404  insert_into_buffer(&ctx->out_ifc_list[ifc], data, size);
405 
406  result = TRAP_E_OK;
407 
408  } else {
409  /* not enough space */
410 
411 #ifdef BUFFERING_CHECK_HEADERS
412  if (trap_check_buffer_content(ctx->out_ifc_list[ifc].buffer, ctx->out_ifc_list[ifc].buffer_index) != 0) {
413  VERBOSE(CL_ERROR, "Buffer is not valid.");
414  }
415 #endif
416 
417  DEBUG_BUF(VERBOSE(CL_VERBOSE_LIBRARY, "sending %"PRIu32" B from %p", ctx->out_ifc_list[ifc].buffer_index, ctx->out_ifc_list[ifc].buffer));
418 
419 #ifdef BUFFERING_CREATE_DUMPS
420  char *n = NULL;
421  if (asprintf(&n, "store-buffers-dump%04"PRIu64, ctx->counter_send_buffer[ifc]) != -1) {
422  mkdir(n, 0700);
423  ctx->out_ifc_list[ifc].create_dump(ctx->out_ifc_list[ifc].priv, ifc, n);
424  free(n);
425  }
426 #endif
427 
428  if (ctx->out_ifc_list[ifc].bufferswitch == 0) {
429  insert_into_buffer(&ctx->out_ifc_list[ifc], data, size);
430  }
431 
432  ctx->out_ifc_list[ifc].buffer_occupied = 1;
433  trap_buffer_header_t *h = (trap_buffer_header_t *) ctx->out_ifc_list[ifc].buffer_header;
434  h->data_length = htonl(ctx->out_ifc_list[ifc].buffer_index);
435  result = ctx->out_ifc_list[ifc].send(ctx->out_ifc_list[ifc].priv, ctx->out_ifc_list[ifc].buffer_header,
436  ctx->out_ifc_list[ifc].buffer_index + sizeof(trap_buffer_header_t), timeout);
437 
438  /* if the buffer was successfully sent OR we have no client: */
439  if (result == TRAP_E_OK || result == TRAP_E_IO_ERROR) {
440  if (result == TRAP_E_OK) {
441  /*
442  * buffer was successfully sent but we still have current message pending/not stored
443  * it will be the first message in buffer
444  */
445  ctx->counter_send_buffer[ifc]++;
446  } else {
447  /* we had no client but we can propagate either OK or TIMEOUT: */
448  result = TRAP_E_TIMEOUT;
449  }
450  /* buffer will be cleaned */
451  ctx->out_ifc_list[ifc].buffer_index = 0;
452  ctx->out_ifc_list[ifc].buffer_occupied = 0;
453  /* buffer was successfully sent but we still have current message pending/not stored
454  * it will be the first message in buffer */
455  if (ctx->out_ifc_list[ifc].bufferswitch == 1) {
456  insert_into_buffer(&ctx->out_ifc_list[ifc], data, size);
457  }
458  } else {
459  if (result == TRAP_E_TIMEOUT) {
460  ctx->counter_dropped_message[ifc]++;
461  }
462  if (trap_ctx_get_client_count(ctx, ifc) == 0) {
463  ctx->out_ifc_list[ifc].buffer_occupied = 0;
464  }
465  }
466  }
467 
468 fn_exit:
469  pthread_mutex_unlock(&ctx->out_ifc_list[ifc].ifc_mtx);
470  return result;
471 }

482 /**
483  * Function of reader-thread.
484  *
485  * \param[in] arg struct reader_threads_arg with context and thread id - used as the index in ctx->in_ifc_list, in_ifc_results
486  * \return NULL
487  */
488 void *reader_threads_fn(void *arg)
489 {
490  struct reader_threads_arg *argdata = (struct reader_threads_arg *) arg;
491  trap_ctx_priv_t *ctx = NULL;
492  int thread_id;
493  int retval;
494 #ifdef DISABLE_BUFFERING
495  uint32_t recvsize = 0;
496 #endif
497 
498  if (argdata == NULL) {
499  pthread_exit(NULL);
500  }
501  ctx = argdata->ctx;
502  thread_id = argdata->thread_index;
503  do {
504  sem_wait(&ctx->reader_threads[thread_id].sem);
505  if (ctx->terminated == 1) {
506  break;
507  }
508  /* call recv of my IFC and let it store results into multi-result array */
509 #ifndef DISABLE_BUFFERING
510  retval = trap_read_from_buffer(ctx, thread_id, (const void **) &ctx->in_ifc_results[thread_id].message,
511  &ctx->in_ifc_results[thread_id].message_size,
512  ctx->get_data_timeout);
513 #else
514  retval = ctx->in_ifc_list[thread_id].recv(ctx->in_ifc_list[thread_id].priv,
515  ctx->in_ifc_list[thread_id].buffer,
516  &recvsize,
517  ctx->get_data_timeout);
518  /* if sender uses buffering, we are loosing data! in addition, data can be corrupted in this case!!! */
519  ctx->in_ifc_results[thread_id].message_size = (uint16_t) recvsize;
520  ctx->in_ifc_results[thread_id].message = ctx->in_ifc_list[thread_id].buffer;
521 #endif
522 
523  ctx->in_ifc_results[thread_id].result_code = retval;
524  pthread_mutex_lock(&ctx->mut_sem_collector);
525  ctx->readers_count--;
526  if (ctx->readers_count == 0) {
527  /* the last reader wakes collector */
528  retval = sem_post(&ctx->sem_collector);
529  if (retval != 0) {
530  VERBOSE(CL_ERROR, "Waking up collector thread of multiread function failed. (%d)", retval);
531  }
532  }
533  pthread_mutex_unlock(&ctx->mut_sem_collector);
534  /* inform collector about finished job */
535 
536  if (ctx->terminated == 1) {
537  break;
538  }
539  } while (1);
540  free(arg);
541  pthread_exit(NULL);
542 }

552 {
553  int i, idx, res;
554  struct out_ifc_timeout_s *out_ifc_timeout = ctx->ifc_autoflush_timeout;
555  idx = 0;
556  for (i = 0; i < ctx->num_ifc_out; i++) {
557  if ((ctx->out_ifc_list[i].timeout != TRAP_NO_AUTO_FLUSH) && (ctx->out_ifc_list[i].bufferswitch != 0)) {
558  out_ifc_timeout[idx].idx = i;
559  // Try lock updating of timeout, it could be changing, do not block
560  res = pthread_mutex_trylock(&ctx->out_ifc_list[i].ifc_mtx);
561  out_ifc_timeout[idx].tm = ctx->out_ifc_list[i].timeout;
562  if (res == 0)
563  pthread_mutex_unlock(&ctx->out_ifc_list[i].ifc_mtx);
564  idx++;
565  }
566  }
567  // All changes updated, set to zero
568  if (pthread_rwlock_wrlock(&ctx->context_lock) != 0) {
569  VERBOSE(CL_ERROR, "Locking of context failed. %s", __func__);
570  }
571  ctx->ifc_change = 0;
572  pthread_rwlock_unlock(&ctx->context_lock);
573  return idx;
574 }

576 /**
577  * Handle the timeouts on output interfaces and flush buffer after timeout is reached.
578  *
579  *
580  * @return NULL
581  */
582 static void *trap_automatic_flush_thr(void *arg)
583 {
584  int i, n;
585  int64_t usec;
586  trap_ctx_priv_t *ctx = (trap_ctx_priv_t *) arg;
587 
588  n = trap_init_ifcs_timeouts(ctx);
589 
590  while (1) {
591  if (pthread_rwlock_rdlock(&ctx->context_lock) != 0) {
592  VERBOSE(CL_ERROR, "Locking of context failed. %s", __func__);
593  break;
594  }
595  if (ctx->terminated == 1) {
596  pthread_rwlock_unlock(&ctx->context_lock);
597  break;
598  }
599  // Checking if automatic flushing or buffering was changed or disabled
600  if (ctx->ifc_change == 1) {
601  pthread_rwlock_unlock(&ctx->context_lock);
602  n = trap_init_ifcs_timeouts(ctx);
603  } else {
604  pthread_rwlock_unlock(&ctx->context_lock);
605  }
606 
607  // Sort array by timeout if we have more than one output interface
608  if (n != 0) {
609  if (n > 1) {
610  qsort(ctx->ifc_autoflush_timeout, n, sizeof(struct out_ifc_timeout_s), compare_timeouts);
611  }
612  usec = ctx->ifc_autoflush_timeout[0].tm;
613  VERBOSE(CL_VERBOSE_LIBRARY, "Autoflush thread is going to sleep for %ld microseconds.", usec);
614  if (sleep(usec/1000000) != 0) {
615  if (pthread_rwlock_rdlock(&ctx->context_lock) != 0) {
616  VERBOSE(CL_ERROR, "Locking of context failed. %s", __func__);
617  break;
618  }
619  if (ctx->terminated == 1) {
620  pthread_rwlock_unlock(&ctx->context_lock);
621  break;
622  }
623  pthread_rwlock_unlock(&ctx->context_lock);
624  }
625  if (usleep(usec%1000000) == -1) {
626  if (pthread_rwlock_rdlock(&ctx->context_lock) != 0) {
627  VERBOSE(CL_ERROR, "Locking of context failed. %s", __func__);
628  break;
629  }
630  if (ctx->terminated == 1) {
631  pthread_rwlock_unlock(&ctx->context_lock);
632  break;
633  }
634  pthread_rwlock_unlock(&ctx->context_lock);
635  }
636 
637  // Check all interfaces if timeout has elapsed, otherwise break
638  for (i = 0; i < n; i++) {
639  ctx->ifc_autoflush_timeout[i].tm -= usec;
640  if (ctx->ifc_autoflush_timeout[i].tm == 0) {
641  pthread_mutex_lock(&ctx->out_ifc_list[i].ifc_mtx);
642  if (ctx->out_ifc_list[ctx->ifc_autoflush_timeout[i].idx].bufferflush == 0) {
643  pthread_mutex_unlock(&ctx->out_ifc_list[i].ifc_mtx);
644  // No event on the interface, flushing the buffer
645  trap_ctx_send_flush((trap_ctx_t *) ctx, i);
646  ctx->counter_autoflush[ctx->ifc_autoflush_timeout[i].idx]++;
647  }
648  else {
649  // Buffer was sent before timeout has elapsed, no need to flush the buffer
650  ctx->out_ifc_list[ctx->ifc_autoflush_timeout[i].idx].bufferflush = 0;
651  pthread_mutex_unlock(&ctx->out_ifc_list[i].ifc_mtx);
652  }
653  // Updating of timeout, it could be changing
654  pthread_mutex_lock(&ctx->out_ifc_list[i].ifc_mtx);
655  ctx->ifc_autoflush_timeout[i].tm = ctx->out_ifc_list[ctx->ifc_autoflush_timeout[i].idx].timeout;
656  pthread_mutex_unlock(&ctx->out_ifc_list[i].ifc_mtx);
657  } else {
658  continue;
659  }
660  }
661  } else {
662  // Sleep for defined time, default 2 seconds
663  if (sleep(TRAP_NO_IFC_SLEEP)) {
664  break;
665  }
666  }
667  }
668 
669  pthread_exit(NULL);
670 }

681 /** Initialization function.
682  * Create and initialize all interfaces. This function parses command-line
683  * arguments; it extracts arguments it needs to set up interfaces and returns
684  * the rest (argc and argv are modified).
685  * @param[in,out] argc Pointer to number of command-line arguments.
686  * @param[in,out] argv Command-line arguments.
687  * @param[out] ifc_spec Structure with specification of interface types and
688  * their parameters.
689  * @return Error code (0 on success)
690  */
691 int trap_parse_params(int *argc, char **argv, trap_ifc_spec_t *ifc_spec)
692 {
693  uint32_t i, j, ifc_count = 0;
694  char *ifc_spec_str = NULL;
695  char *ifc_type = NULL;
696  char *p;
697  int rv = TRAP_E_OK;
698 
699  if (ifc_spec == NULL) {
700  VERBOSE(CL_ERROR, "Bad pointer 'ifc_spec' passed to trap_parse_params().");
701  return TRAP_E_BAD_FPARAMS;
702  }
703 
704  /* initialization of ifc_spec */
705  memset(ifc_spec, 0, sizeof(trap_ifc_spec_t));
706 
707  // Look for -h (or --help) parameter
708  for (i = 0; i < *argc; i++) {
709  if (strcmp(argv[i], "-h") == 0 || strcmp(argv[i], "--help") == 0) {
710  if (i + 1 < *argc && ((strcmp(argv[i + 1], "trap") == 0) || (strcmp(argv[i + 1], "1") == 0))) {
711  trap_help_spec = 1;
712  }
713  // Remove parameter from argv and break
714  for (j = i; j + 1 < *argc; j++) {
715  argv[j] = argv[j + 1];
716  }
717  *argc -= 1;
718  return TRAP_E_HELP;
719  }
720  }
721 
722  // Extract -i parameter (interface specifier)
723  for (i = 0; i < *argc-1; i++) {
724  if (argv[i] && strcmp(argv[i], "-i") == 0) {
725  ifc_spec_str = argv[i+1];
726  // Remove parameters from argv
727  for (j = i; j + 2 < *argc; j++) {
728  argv[j] = argv[j + 2];
729  }
730  *argc -= 2;
731  break;
732  }
733  }
734  if (ifc_spec_str == NULL) {
735  //return trap_errorf(ctx, TRAP_E_BADPARAMS, "Interface specifier (option -i) not found.");
736  trap_last_error = TRAP_E_BADPARAMS;
737  trap_last_error_msg = "Interface specifier (option -i) not found.";
738  return TRAP_E_BADPARAMS;
739  }
740 
741  // Extract verbose level parameter (-v, -vv, -vvv)
742  for (i = 0; i < *argc; i++) {
743  // If param matches -v, -vv or -vvv, set verbosity level
744  if (strcmp(argv[i], "-v") == 0) {
745  trap_set_verbose_level(0);
746  } else if (strcmp(argv[i], "-vv") == 0) {
747  trap_set_verbose_level(1);
748  } else if (strcmp(argv[i], "-vvv") == 0) {
749  trap_set_verbose_level(2);
750  } else {
751  continue;
752  }
753  // Remove parameter from argv and break
754  for (j = i; j + 1 < *argc; j++) {
755  argv[j] = argv[j + 1];
756  }
757  *argc -= 1;
758  break;
759  }
760 
761  /* count the number of IFC parameters, the format is:
762  * type1:param1:param1_2,type2:,type3:param3 */
763  p = ifc_spec_str;
764  while (p && *p != '\0') {
765  p = strchr(p, TRAP_IFC_DELIMITER);
766  if (p != NULL && *p == TRAP_IFC_DELIMITER) {
767  p++;
768  }
769  ifc_count++;
770  }
771 
772  ifc_spec->types = calloc(ifc_count + 1, sizeof(*ifc_spec->types));
773  if (ifc_spec->types == NULL) {
774  return TRAP_E_MEMORY;
775  }
776  ifc_spec->params = calloc(ifc_count, sizeof(char *));
777  if (ifc_spec->params == NULL) {
778  free(ifc_spec->types);
779  ifc_spec->types = NULL;
780  return TRAP_E_MEMORY;
781  }
782 
783  p = ifc_spec_str;
784  for (i = 0; i < ifc_count; i++) {
785  /* set IFC type and skip to params */
786  if (p != NULL) {
787  ifc_spec->types[i] = p[0];
788  if (strlen(p) >= 2 && p[1] == TRAP_IFC_PARAM_DELIMITER) {
789  /* there is a param string to parse */
790  p += 2;
791  p = get_param_by_delimiter(p, &ifc_spec->params[i], TRAP_IFC_DELIMITER);
792  } else {
793  /* param was not passed, set an empty one */
794  ifc_spec->params[i] = strdup("");
795  }
796  } else {
797  /* unexpected error, p shouldn't be NULL */
798  VERBOSE(CL_ERROR, "Bad IFC_SPEC '%s'. See -h trap for help.", ifc_spec_str);
799  ifc_spec->params[i] = strdup("");
800  }
801  if (ifc_spec->params[i] == NULL) {
802  VERBOSE(CL_ERROR, "Allocation failed.");
803  rv = TRAP_E_MEMORY;
804  goto clean_on_fail;
805  }
806  }
807 
808  /* check for unsupported IFCs */
809  for (i = 0; ifc_spec->types[i] != 0; ++i) {
810  for (ifc_type = trap_ifc_type_supported; *ifc_type != 0; ++ifc_type) {
811  if (*ifc_type == ifc_spec->types[i]) {
812  break;
813  }
814  }
815  if (*ifc_type == 0) {
816  /* not found */
817  VERBOSE(CL_ERROR, "Unsupported IFC type '%c'.", ifc_spec->types[i]);
818  rv = TRAP_E_BADPARAMS;
819  goto clean_on_fail;
820  }
821  }
822 
823  trap_last_error = TRAP_E_OK;
824  trap_last_error_msg = default_err_msg[TRAP_E_OK];
825  return TRAP_E_OK;
826 
827 clean_on_fail:
828  for (i = 0; i < ifc_count; i++) {
829  /* set IFC type and skip to params */
830  if (ifc_spec->params[i] != NULL) {
831  free(ifc_spec->params[i]);
832  ifc_spec->params[i] = NULL;
833  }
834  }
835  if (ifc_spec->types != NULL) {
836  free(ifc_spec->types);
837  }
838  if (ifc_spec->params) {
839  free(ifc_spec->params);
840  }
841  memset(ifc_spec, 0, sizeof(trap_ifc_spec_t));
842  return rv;
843 }
